'use strict';

const {
  SPRITE_SIZE_IN_BYTES
} = require('../constants');

const Entity = require('./entity');

/**
 * Generic example.
 * @type {Object}
 */
class Sprite extends Entity {
  /**
   * Generic template for virtual objects.
   * @param  {Object} [data={}] Pass an object to use.
   * @return {Entity}           Instance of the {@link Entity}.
   */
  constructor (data = {}) {
    super(data);

    // allow this entity to be run without the new keyword
    if (!(this instanceof Sprite)) return new Sprite(data);

    this.settings = Object.assign({
      frames: 32
    }, data);

    // set internal properties
    this.sheet = Buffer.alloc(SPRITE_SIZE_IN_BYTES);
    this.frame = Buffer.alloc(SPRITE_SIZE_IN_BYTES / this.settings.frames);

    // configure defaults
    this.data = Object.assign({
      frame: this.frame.toString('hex')
    }, data);

    this.state = Object.assign({
      frame: this.data.frame,
      frames: {}
    }, {});

    this.commit();

    // return instance
    return this;
  }

  _loadSheet (path) {
    // TODO: load sheet here (@melnx)
    this.sheet.write(null);
  }

  tick () {
    
  }

  render () {
    return `<fabric-sprite>${JSON.stringify(this.data)}</fabric-sprite>`;
  }
}

module.exports = Sprite;
